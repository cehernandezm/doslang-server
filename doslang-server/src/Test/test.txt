Program Hola;
{*asda

s*}
type
	nodo = record
			nombre:String;
			siguiente:nodo;
			anterior:nodo;
	end;

var
primero : nodo;
ultimo : nodo;
temp : nodo;

Procedure guardar(m:String);
	var nuevo:nodo;
	BEGIN
		nuevo := malloc(sizeof(primero));
		nuevo.nombre := m;
		nuevo.siguiente := NIL;
		nuevo.anterior := NIL;
		if(primero = NIL) then 
		BEGIN 
			primero := nuevo;
			ultimo := nuevo;
		END;
		ELSE
		BEGIN
			ultimo.siguiente := nuevo;
			nuevo.anterior := ultimo;
			ultimo := nuevo;
		END;
	END;

BEGIN

primero := NIL;
ultimo := NIL;

guardar("hola");
guardar("Hola2");

temp := primero;
while(temp <> NIL) DO
BEGIN
	writeln(temp.nombre);
	temp := temp.siguiente;
END;


END.
