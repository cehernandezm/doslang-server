function ackerman(m: integer,n: integer): integer;
VAR 
ackerman : integer;

begin
    if( m = 0) THEN ackerman := n + 1;
    else if(m > 0 AND n <= 0) THEN ackerman := ackerman(m - 1, 1);
    else ackerman := ackerman(m - 1, ackerman(m, n - 1));
        
end;
