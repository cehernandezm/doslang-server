program arbolavl;
type
	Node = Record
		key : integer;
		left : Node;
		right : Node;
		height : integer;
	end;
var
raiz : Node; (* Raiz del arbol *)
(* Altura del nodo actual *)
procedure insertar(numero:Integer;root:node);
var nodo:Node;
	
begin
	nodo := malloc(sizeof(node));
	nodo.key := numero;

	if(root = NIL) then
	begin
		root := nodo;
		exit;
	end;
	else if(numero < root.key) then insertar(numero,root.left);
	else insertar(numero,root.right);

END;

procedure preorden(root:Node);
begin
	if(root = NIL) then exit;
	writeln(root.key);
	preorden(root.left);
	preorden(root.right);
end;

begin
	raiz := NIL;
	insertar(13,raiz);
	insertar(14,raiz);
	insertar(15,raiz);
	
	write("PreOrden:");
	preorden(raiz);
	
end.
