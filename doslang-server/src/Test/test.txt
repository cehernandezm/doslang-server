Type

		hola : String;
		siguiente : Nodo;
		anterior : Nodo; 
		END;

primero : Nodo = NIL;
ultimo : Nodo = NIL;
temp : Nodo;

procedure guardar(m:String);
var nuevo: Nodo;
BEGIN
nuevo := malloc(3);
nuevo.hola := m;

BEGIN
	primero := nuevo;
	ultimo := nuevo;
END;
else 
BEGIN
	ultimo.siguiente := nuevo;
	nuevo.anterior := ultimo;
	ultimo := nuevo;
END;
END;
